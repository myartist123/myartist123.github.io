    FPGA-定时器

    一、需求分析

    1.定时器的时间为15 s，按递减方式进行计时，每隔一秒钟定时器减1；

    2.定时器的时间以数字形式显示在数码管上；

    3.设置两个外部控制开关，控制定时器的直接复位、启动计时，暂停/连续计时；

    4.当计时器递减到0时，定时器保持不变，同时发出信号提示灯（4个led灯点亮，表示15s时间已到）

    六、编写RTL代码

    用notepad++软件

    module Timer(
    input wire clk,     // 时钟信号
    input wire reset,   // 复位信号
    output reg [3:0] time_remaining,  // 剩余时间信号，4位二进制数表示剩余秒数
    output reg trigger  // 触发信号，在计时器归零时触发
);

// 内部计数器
reg [3:0] counter;

always @(posedge clk or posedge reset) begin
    if (reset) begin
        // 在复位信号下，将计数器和剩余时间重置为初始值
        counter <= 4'b1111;  // 15秒
        time_remaining <= counter;
        trigger <= 1'b0;     // 触发信号清零
    end else begin
        // 在时钟信号下，递减计时
        if (counter > 0) begin
            counter <= counter - 1;
            time_remaining <= counter;
        end else begin
            // 当计时器归零时，触发信号置位
            trigger <= 1'b1;
        end
    end
end

endmodule

module SevenSegmentDisplay(
    input wire [3:0] number,    // 输入数字信号，4位二进制数表示
    output reg [6:0] segments   // 输出数码管分段信号，共阴极
);

// 数码管显示表，对应的是共阴极七段数码管的真值表
// 顺序为ABCDEFG，1表示点亮，0表示熄灭
reg [6:0] segment_table [0:9] = {
    7'b0000001,  // 0
    7'b1001111,  // 1
    7'b0010010,  // 2
    7'b0000110,  // 3
    7'b1001100,  // 4
    7'b0100100,  // 5
    7'b0100000,  // 6
    7'b0001111,  // 7
    7'b0000000,  // 8 (留空)
    7'b0000100   // 9
};

// 根据输入数字选择对应的数码管显示表项
always @* begin
    case (number)
        4'b0000: segments = segment_table[0];
        4'b0001: segments = segment_table[1];
        4'b0010: segments = segment_table[2];
        4'b0011: segments = segment_table[3];
        4'b0100: segments = segment_table[4];
        4'b0101: segments = segment_table[5];
        4'b0110: segments = segment_table[6];
        4'b0111: segments = segment_table[7];
        4'b1000: segments = segment_table[8];  // 显示空白
        4'b1001: segments = segment_table[9];
        default: segments = 7'b1111111;  // 默认显示错误
    endcase
end

endmodule

module ExternalControl(
    input wire reset_switch,      // 复位控制开关
    input wire start_pause_switch,// 启动/暂停控制开关
    output reg enable_clock       // 时钟使能信号
);

// 初始状态下，时钟使能信号为0，即计时器停止
assign enable_clock = 1'b0;

// 使用状态机来控制计时器的行为
// 状态机包含两个状态：停止和运行
// 当启动/暂停控制开关切换时，状态机的状态会相应地改变
reg [1:0] state = 2'b00;

always @(negedge start_pause_switch or posedge reset_switch) begin
    if (reset_switch) begin
        // 复位控制开关被触发时，状态机复位到停止状态
        state <= 2'b00;
        enable_clock <= 1'b0;
    end else begin
        // 启动/暂停控制开关被触发时，状态机状态切换
        case (state)
            2'b00: begin
                // 当前状态为停止状态，切换到运行状态
                if (start_pause_switch) begin
                    state <= 2'b01;
                    enable_clock <= 1'b1; // 启动计时器
                end
            end
            2'b01: begin
                // 当前状态为运行状态，切换到停止状态
                if (start_pause_switch) begin
                    state <= 2'b00;
                    enable_clock <= 1'b0; // 暂停计时器
                end
            end
            default: state <= 2'b00; // 默认状态为停止状态
        endcase
    end
end

endmodule

module LEDIndicator(
    input wire trigger,  // 触发信号，计时器归零时触发
    output wire [3:0] leds // LED提示灯信号，4个LED灯
);

// 初始状态下，LED提示灯信号为0，即灯处于熄灭状态
assign leds = 4'b0000;

// 当收到触发信号时，点亮LED提示灯
always @(posedge trigger) begin
    leds <= 4'b1111; // 点亮所有4个LED灯
end

endmodule
