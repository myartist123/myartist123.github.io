    数据集：RML2016.10a

    1. 数据预处理和加载：
      • 将数据集分为训练集、验证集和测试集。
      • 使用DataLoader来加载数据。
    2. 定义ResNet34模型：
      • 使用PyTorch的预训练模型，并调整最后一层以适应11种调制方式。
    3. 训练模型：
      • 定义训练循环，包括损失函数和优化器。
      • 实现验证循环以评估模型在验证集上的表现。
    4. 评估和可视化结果：
      • 计算不同信噪比下的平均准确率并绘制折线图。
      • 计算不同信噪比下的调制方式准确率并绘制折线图。
      • 生成混淆矩阵以展示不同调制方式的分类效果。
    import os
import pickle  # 添加这行来导入pickle模块
import time
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
import torch.utils.data as data
import torchvision.models as models
import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix, ConfusionMatrixDisplay
from sklearn.model_selection import train_test_split
from tqdm import tqdm
import copy  # 添加这行来导入copy模块

# 数据集路径
dataset_path = 'RML2016.10a_dict.pkl'
model_save_path = 'resnet34_model.pth'

# 加载数据集
def load_data(dataset_path):
    with open(dataset_path, 'rb') as f:
        Xd = pickle.load(f, encoding='latin1')
    snrs, mods = map(lambda j: sorted(list(set(map(lambda x: x[j], Xd.keys())))), [1, 0])
    X = []
    lbl = []
    for mod in mods:
        for snr in snrs:
            X.append(Xd[(mod, snr)])
            for i in range(Xd[(mod, snr)].shape[0]):
                lbl.append((mod, snr))
    X = np.vstack(X)
    return X, lbl, snrs, mods

# 数据集分割
def split_data(X, lbl, test_size=0.2, val_size=0.1, random_state=42):
    n_samples = len(lbl)
    n_test = int(n_samples * test_size)
    n_val = int(n_samples * val_size)

    X_train_val, X_test, lbl_train_val, lbl_test = train_test_split(X, lbl, test_size=n_test, random_state=random_state)
    X_train, X_val, lbl_train, lbl_val = train_test_split(X_train_val, lbl_train_val, test_size=n_val, random_state=random_state)

    return (X_train, lbl_train), (X_val, lbl_val), (X_test, lbl_test)

# 定义数据集类
class ModulationDataset(data.Dataset):
    def __init__(self, X, lbl, mods, snrs, transform=None):
        self.X = X
        self.lbl = lbl
        self.mods = mods
        self.snrs = snrs
        self.transform = transform

    def __len__(self):
        return len(self.lbl)

    def __getitem__(self, idx):
        sample = self.X[idx]
        sample = np.expand_dims(sample, axis=1)  # 增加一个维度
        label = self.mods.index(self.lbl[idx][0])
        snr = self.snrs.index(self.lbl[idx][1])
        if self.transform:
            sample = self.transform(sample)
        return sample, label, snr

# 数据加载器
def get_dataloaders(X_train, lbl_train, X_val, lbl_val, X_test, lbl_test, mods, snrs, batch_size=128):
    train_dataset = ModulationDataset(X_train, lbl_train, mods, snrs, transform=torch.tensor)
    val_dataset = ModulationDataset(X_val, lbl_val, mods, snrs, transform=torch.tensor)
    test_dataset = ModulationDataset(X_test, lbl_test, mods, snrs, transform=torch.tensor)

    train_loader = data.DataLoader(train_dataset, batch_size=batch_size, shuffle=True)
    val_loader = data.DataLoader(val_dataset, batch_size=batch_size, shuffle=False)
    test_loader = data.DataLoader(test_dataset, batch_size=batch_size, shuffle=False)

    return train_loader, val_loader, test_loader

# 定义ResNet34模型
class ResNet34(nn.Module):
    def __init__(self, num_classes=11):
        super(ResNet34, self).__init__()
        self.model = models.resnet34(pretrained=True)
        self.model.conv1 = nn.Conv2d(2, 64, kernel_size=7, stride=2, padding=3, bias=False)
        self.model.fc = nn.Linear(512, num_classes)

    def forward(self, x):
        return self.model(x)

# 训练模型
def train_model(model, train_loader, val_loader, criterion, optimizer, num_epochs=10, device='cuda', model_save_path='resnet34_model.pth'):
    best_model_wts = copy.deepcopy(model.state_dict())
    best_acc = 0.0

    print(f'Training on {device}')

    for epoch in range(num_epochs):
        print(f'Epoch {epoch}/{num_epochs - 1}')
        print('-' * 10)

        epoch_start = time.time()

        for phase in ['train', 'val']:
            if phase == 'train':
                model.train()
                dataloader = train_loader
            else:
                model.eval()
                dataloader = val_loader

            running_loss = 0.0
            running_corrects = 0

            for inputs, labels, _ in tqdm(dataloader):
                inputs = inputs.to(device).float()
                labels = labels.to(device)

                optimizer.zero_grad()

                with torch.set_grad_enabled(phase == 'train'):
                    outputs = model(inputs)
                    _, preds = torch.max(outputs, 1)
                    loss = criterion(outputs, labels)

                    if phase == 'train':
                        loss.backward()
                        optimizer.step()

                running_loss += loss.item() * inputs.size(0)
                running_corrects += torch.sum(preds == labels.data)

            epoch_loss = running_loss / len(dataloader.dataset)
            epoch_acc = running_corrects.double() / len(dataloader.dataset)

            print(f'{phase} Loss: {epoch_loss:.4f} Acc: {epoch_acc:.4f}')

            if phase == 'val' and epoch_acc > best_acc:
                best_acc = epoch_acc
                best_model_wts = copy.deepcopy(model.state_dict())

        epoch_time = time.time() - epoch_start
        print(f'Time for epoch {epoch}: {epoch_time:.2f} seconds')

        torch.save(model.state_dict(), model_save_path)

    print(f'Best val Acc: {best_acc:4f}')
    model.load_state_dict(best_model_wts)
    return model

# 评估模型
def evaluate_model(model, dataloader, device='cuda'):
    model.eval()
    running_corrects = 0

    all_preds = []
    all_labels = []
    all_snrs = []

    print(f'Evaluating on {device}')

    for inputs, labels, snrs in tqdm(dataloader):
        inputs = inputs.to(device).float()
        labels = labels.to(device)

        with torch.set_grad_enabled(False):
            outputs = model(inputs)
            _, preds = torch.max(outputs, 1)

        running_corrects += torch.sum(preds == labels.data)
        all_preds.extend(preds.cpu().numpy())
        all_labels.extend(labels.cpu().numpy())
        all_snrs.extend(snrs.numpy())

    acc = running_corrects.double() / len(dataloader.dataset)
    return acc, all_preds, all_labels, all_snrs

# 绘制不同信噪比下的平均准确率折线图
def plot_avg_accuracy(snrs, accuracies):
    plt.figure()
    plt.plot(snrs, accuracies)
    plt.xlabel('SNR (dB)')
    plt.ylabel('Accuracy')
    plt.title('Average Accuracy vs SNR')
    plt.grid(True)
    plt.show()

# 绘制不同信噪比下的调制方式准确率折线图
def plot_mod_accuracy(snrs, mod_accuracies, mods):
    for i, mod in enumerate(mods):
        plt.plot(snrs, mod_accuracies[i], label=mod)
    plt.xlabel('SNR (dB)')
    plt.ylabel('Accuracy')
    plt.title('Modulation Accuracy vs SNR')
    plt.legend()
    plt.grid(True)
    plt.show()

# 绘制混淆矩阵
def plot_confusion_matrix(y_true, y_pred, mods):
    cm = confusion_matrix(y_true, y_pred)
    disp = ConfusionMatrixDisplay(confusion_matrix=cm, display_labels=mods)
    disp.plot(cmap=plt.cm.Blues)
    plt.show()

# 主函数
def main():
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

    X, lbl, snrs, mods = load_data(dataset_path)
    
    (X_train, lbl_train), (X_val, lbl_val), (X_test, lbl_test) = split_data(X, lbl)

    train_loader, val_loader, test_loader = get_dataloaders(X_train, lbl_train, X_val, lbl_val, X_test, lbl_test, mods, snrs)

    model = ResNet34(num_classes=len(mods)).to(device)
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=0.001)

    if os.path.exists(model_save_path):
        print("Loading saved model...")
        model.load_state_dict(torch.load(model_save_path))
    else:
        model = train_model(model, train_loader, val_loader, criterion, optimizer, num_epochs=10, device=device, model_save_path=model_save_path)

    acc, all_preds, all_labels, all_snrs = evaluate_model(model, test_loader, device=device)
    print(f'Test Accuracy: {acc:.4f}')

    snr_acc = {snr: [] for snr in snrs}
    for snr, label, pred in zip(all_snrs, all_labels, all_preds):
        if snr in snr_acc:
            snr_acc[snr].append(label == pred)
        else:
            print(f'Warning: SNR {snr} not found in snr_acc dictionary.')

    avg_accuracy = [np.mean(snr_acc[snr]) for snr in snrs]
    plot_avg_accuracy(snrs, avg_accuracy)

    mod_snr_acc = {mod: {snr: [] for snr in snrs} for mod in mods}
    for snr, label, pred in zip(all_snrs, all_labels, all_preds):
        mod_snr_acc[mods[label]][snr].append(label == pred)

    mod_accuracy = [[np.mean(mod_snr_acc[mod][snr]) for snr in snrs] for mod in mods]
    plot_mod_accuracy(snrs, mod_accuracy, mods)

    plot_confusion_matrix(all_labels, all_preds, mods)

if __name__ == '__main__':
    main()

from sklearn.model_selection import train_test_split
from tqdm import tqdm

    return X_test, lbl_test

    dataloader = data.DataLoader(dataset, batch_size=batch_size, shuffle=shuffle)

# 主函数
def main():
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

    X, lbl, snrs, mods = load_data(dataset_path)
    X_test, lbl_test = split_data(X, lbl)

    test_loader = get_dataloader(X_test, lbl_test, mods, snrs, shuffle=False)

    model = ResNet34(num_classes=len(mods)).to(device)

    if os.path.exists(model_save_path):
        print("Loading saved model...")
        model.load_state_dict(torch.load(model_save_path))
    else:
        raise FileNotFoundError(f'Model file not found at {model_save_path}. Please train the model first.')

    acc, all_preds, all_labels, all_snrs = evaluate_model(model, test_loader, device=device)
    print(f'Test Accuracy: {acc:.4f}')

    snr_acc = {snr: [] for snr in range(len(snrs))}
    for snr, label, pred in zip(all_snrs, all_labels, all_preds):
        snr_acc[snr].append(label == pred)

    avg_accuracy = [np.mean(snr_acc[snr]) if snr_acc[snr] else 0 for snr in range(len(snrs))]
    plot_avg_accuracy(snrs, avg_accuracy)

    mod_snr_acc = {mod: {snr: [] for snr in range(len(snrs))} for mod in range(len(mods))}
    for snr, label, pred in zip(all_snrs, all_labels, all_preds):

    mod_accuracy = [[np.mean(mod_snr_acc[mod][snr]) if mod_snr_acc[mod][snr] else 0 for snr in range(len(snrs))] for mod in range(len(mods))]
